void start_7()
{
	uint a, b = 1000;

	{
		P0 = 0xff;
		delay_us(b--);
		P0 = 0x00;
		delay_us(a);

	b = 1000;
	for (a = 1; a < 1000; a++)
	{
		P0 = 0x00;
		delay_us(b--);
		P0 = 0xff;

void start_8()
{
	uchar a, b, c, d, e, f, i;
	c = 0x7f;
	d = 0x7f;

	e = 0xfe;
	f = 0xfe;

	for (a = 1; a < 5; a++)
	{
		P0 = c & e;

		b = d;
		for (i = 0; i < (8 - a); i++)
		{
			P0 = b & c & e;
			delay_us(50000);
			b = _cror_(b, 1);
		}

		e = e << 1;
		f = _crol_(f, 1);
		P0 = c & e;

		b = f;
		for (i = 0; i < (7 - a); i++)
		{
			P0 = b & c & e;
			delay_us(50000);
			b = _crol_(b, 1);
		}

		c = c >> 1;
		d = _cror_(d, 1);
	}
}

void main()

			start_8();

	{

			start_8();
		}
	}
}

#include <STC8A.h>
#define uchar unsigned char
#define uint unsigned int

uchar table[] = {0xc0, 0xf9, 0xa4, 0xb0, 0x99, 0x92, 0x82, 0xf8, 0x80, 0x90, 0xbf};
uchar select[] = {0xfe, 0xfd, 0xfb, 0xf7, 0xef, 0xdf, 0xbf, 0x7f};

sbit k1 = P3 ^ 2;
sbit k2 = P3 ^ 3;
sbit k3 = P3 ^ 4;
sbit k4 = P3 ^ 5;
sbit k5 = P4 ^ 0;
sbit beep = P4 ^ 2;

bit flag1, flag2, flag3, flag4, flag6, flag7;
char flag5;

char hour, minute, second, year, month, day, alarm_hour, alarm_minute;

void delay()
{
	uint i = 1000;
	while (i--)
		;
}

void delay_us(uint us)
{
	uint a;
	uchar b;
	for (b = 0; b < 2; b++)
		for (a = 0; a < us; a++)
			;
}

void BEEP(uint us)
{
	beep = ~beep;
	delay_us(us);
	beep = ~beep;
}

void init()
{
	P0M1 = 0x00;
	P0M0 = 0xff;
	P1M1 = 0x00;
	P1M0 = 0xff;
	P2M1 = 0x00;
	P2M0 = 0xff;

	AUXR &= 0x7F;
	TMOD &= 0xF0;
	TL0 = 0xC0;
	TH0 = 0x63;
	TR0 = 1;
	ET0 = 1;
	EA = 1;

	flag5 = 0;

	hour = 0;
	minute = 0;
	second = 0;
	year = 23;
	month = 4;
	day = 21;

	alarm_hour = 12;
	alarm_minute = 0;
}

// 时间显示
void display_time()
{
	P1 = select[0];

// 闹钟报时功能
void alarm()
{
	if (alarm_hour == hour)
	{
		if (alarm_minute == minute)
		{
			BEEP(50000);
		}
	}
}

// 功能切换实现
void function()
{
	if (k5 == 0)
	{
		if (flag6 == 0)
		{
			flag6 = 1;
			BEEP(50000);
			switch (flag5)
			{
			case 0:
				flag5 = 1;
				break;
			case 1:
				flag5 = 2;
				break;
			case 2:
				flag5 = 0;
				break;
			}
		}
	}

	if (k5 == 1)
	{
		flag6 = 0;
	}
}

void main()
{
	init();
	P2 = 0x00;
	while (1)
	{
		if (flag5 == 0)
		{
			// 时间显示
			display_time();
			set_time();
		}
		if (flag5 == 1)
		{
			// 日期显示
			display_date();
			set_date();
		}
		if (flag5 == 2)
		{
			// 闹钟显示
			display_alarm();
			set_alarm();
		}

		function();
		alarm();
	}
}

void timer0() interrupt 1
{
	uchar t0;

	t0++;

	if (t0 == 50)
	{

		second++;
		if (second >= 60)

			second = 0;

		}
	}

	if (hour < 0)

		hour = 0;

		minute = 59;
